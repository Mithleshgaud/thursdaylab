package abstraction;

public abstract class Prime_number
{
	int num,i;
	public void prime() {
		for(num=1;num<=1000;num++) { // loop for 1 to 1000
		
			for(i=2;i<1000;i++) { // division
				if(num%i==0) {  
					break;
				}
			}
			if(i==num) { // number and division value same  it will print if statement
		System.out.print(num );	
			}	
		}
	}

}
